Problem Definition

Write a script that shows the sign (+ or -) of the product of three real numbers without calculating it. Use a sequence of if statements.

JavaScript Code as text

var a = -5;                                                         
var b = 10;                                                          
var c = -5;                                                          
                                                                                 
var countMinus = 0;                                                  
if (a < 0) countMinus++;                                             
if (b < 0) countMinus++;                                             
if (c < 0) countMinus++;                                             
                                                                                 
if (countMinus % 2 === 0) {                                          
    document.writeln("The product sing is +");                       
}                                                                    
else {                                                               
    document.writeln("The product sing is -");                       
}
