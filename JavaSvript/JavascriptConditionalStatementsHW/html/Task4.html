Problem Definition

Sort 3 real values in descending order using nested if statements.

JavaScript Code as text


var a = 6.343;
var b = 8.234;
var c = 6.899;

if (a >= b) {
    if (a >= c) {
        if (b <= c) {
            //swap( b,  c);
            var temp = b;
            b = c;
            c = temp;
        }
    }
    else {
        //swap( a,  c);
        var temp = a;
        a = c;
        c = temp;
        //swap( b,  c);

        b = c;
        c = temp;
    }
}
